231  /* Following methods are guarded by a global critical section. Two of these
232  function will never be invoked in parallel. The first is called when
233  the library is initialised, the second one when it is terminated, i.e.
234  when there are no more open sockets. Either of them can be set to NULL
235  if no specific initialisation/termination is needed. */
236  void (*init) (void);
237  void (*term) (void);
238 
239  /* Each of these functions creates an endpoint and returns the newly
240  created endpoint in 'epbase' parameter. 'hint' is in opaque pointer
241  to be passed to nn_epbase_init(). The epbase object can then be used
242  to retrieve the address to bind/connect to. These functions are guarded
243  by a socket-wide critical section. Two of these function will never be
244  invoked in parallel on the same socket. */
245  int (*bind) (void *hint, struct nn_epbase **epbase);
246  int (*connect) (void *hint, struct nn_epbase **epbase);
