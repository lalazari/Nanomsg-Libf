41 /* This structure represents a node in patricia trie. It's a header to be
42  followed by the array of pointers to child nodes. Each node represents
43  the string composed of all the prefixes on the way from the trie root,
44  including the prefix in that node. */
45 struct nn_trie_node
46 {
47  /* Number of subscriptions to the given string. */
48  uint32_t refcount;
49 
50  /* Number of elements is a sparse array, or NN_TRIE_DENSE_TYPE in case
51  the array of children is dense. */
52  uint8_t type;
53 
54  /* The node adds more characters to the string, compared to the parent
55  node. If there is only a single character added, it's represented
56  directly in the child array. If there's more than one character added,
57  all but the last one are stored as a 'prefix'. */
58  uint8_t prefix_len;
59  uint8_t prefix [NN_TRIE_PREFIX_MAX];
60 
61  /* The array of characters pointing to individual children of the node.
62  Actual pointers to child nodes are stored in the memory following
63  nn_trie_node structure. */
64  union {
65 
66  /* Sparse array means that individual children are identified by
67  characters stored in 'children' array. The number of characters
68  in the array is specified in the 'type' field. */
69  struct {
70  uint8_t children [NN_TRIE_SPARSE_MAX];
71  } sparse;
72 
73  /* Dense array means that the array of node pointers following the
74  structure corresponds to a continuous list of characters starting
75  by 'min' character and ending by 'max' character. The characters
76  in the range that have no corresponding child node are represented
77  by NULL pointers. 'nbr' is the count of child nodes. */
78  struct {
79  uint8_t min;
80  uint8_t max;
81  uint16_t nbr;
82  /* There are 4 bytes of padding here. */
83  } dense;
84  } u;
85 };
