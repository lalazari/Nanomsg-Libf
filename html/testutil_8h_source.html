127 static void NN_UNUSED test_recv_impl (char *file, int line, int sock, char *data)
128 {
129  size_t data_len;
130  int rc;
131  char *buf;
132 
133  data_len = strlen (data);
134  /* We allocate plus one byte so that we are sure that message received
135  has corrent length and not truncated */
136  buf = malloc (data_len+1);
137  alloc_assert (buf);
138 
139  rc = nn_recv (sock, buf, data_len+1, 0);
140  if (rc < 0) {
141  fprintf (stderr, "Failed to recv: %s [%d] (%s:%d)\n",
142  nn_err_strerror (errno),
143  (int) errno, file, line);
144  nn_err_abort ();
145  }
146  if (rc != (int)data_len) {
147  fprintf (stderr, "Received data has wrong length: %d != %d (%s:%d)\n",
148  rc, (int) data_len,
149  file, line);
150  nn_err_abort ();
151  }
152  if (memcmp (data, buf, data_len) != 0) {
153  /* We don't print the data as it may have binary garbage */
154  fprintf (stderr, "Received data is wrong (%s:%d)\n", file, line);
155  nn_err_abort ();
156  }
157 
158  free (buf);
159 }
